##### 예제 2

---

```c
#include<stdio.h>
#include<stdlib.h>

int main(){

	int *pc = NULL;
	
	pc = (char *)malloc(100 * sieof(char));	
    // char 형을 100개 할당한 후 그 주소 값을 포인터 변수 pc에 넣음
    // 배열을 포인터에 넣은 것처럼 사용 가능
    
    for(int i = 0; i < 26; i++){
        
        *(pc + i) = 'a' + i;
    }
}
```
